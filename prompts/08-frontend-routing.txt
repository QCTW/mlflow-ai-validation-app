5. Footer Component (Inline)
   Content: "MLflow AI Validation App - Connect end users with AI engineers"
   Class: .footer

LAYOUT PATTERN:
<Router>
  <div className="app">

        <Route path="/validate/:runId" element={<Validation />} />
      </Routes>
    </main>
    <footer>...</footer>
  </div>
</Router>
